#define LED_PIN D0  // Built-in LED

void setup() {
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  if (Serial.available()) {
    String command = Serial.readStringUntil('\n');
    command.trim();
    
    if (command == "LED_ON") {
      digitalWrite(LED_PIN, HIGH);
    } 
    else if (command == "LED_OFF") {
      digitalWrite(LED_PIN, LOW);
    }
  }
}
